133// These are in one struct because the allocator and the memory used in the
134// base map are linked. You neeed to keep the allocators and the map together
135// because you'll need to manipulate the memory allocated with the same
136// allocator. NOTE: I tried to make this structure flat, using arrays to hold
137// differently typed values, but I ran into a problem. I couldn't know how deep
138// an item was in a structure. To search for a named value, I could look
139// through the array of all the names, but that array didn't have the nesting
140// level of the name. I could add a nesting level, but that could make search
141// times bad for an unsuccessful linear search. I may go back to having a root
142// map, and storing items in tagged unions. I'll probably have to use void* to
143// nest maps in maps. Making maps self referential is difficult otherwise. I
144// had an epiphany. Only the names need to be nested. The values don't. I can
145// store all the blobs and types and values in one array, with indicies
146// referencing it if necessary. I'm not sure how efficient that will be though.
147// I need to think about this further. There's only an extra storage cost if I
148// add an extra pointer to something. BLOBS don't cost extra because I need to
149// store data behind a pointer anyway. A reference to a map might cost extra
150// because that could be an index into an array of maps. Same for arrays. There
151// will be pointers to those, but they could be indexes into an array of
152// arrays.

154 realloc_fn realloc_inst;
155 free_fn free_inst;

157 nvf_num array_num, array_cap;
158 nvf_array *arrays;

160 nvf_num map_num, map_cap;
161 nvf_map *maps;
162 uint8_t init_val;
163} nvf_root;
